## ncnn 代码规范要点

本章遵循 ncnn 风格的 C++ 设计：

### 缓存行对齐 (参考 ncnn 内存优化实践)
```cpp
namespace arm64lab {

// 缓存行对齐包装器 (ncnn 风格)
template<typename T>
struct alignas(64) CacheLineAligned {
    T value;

    CacheLineAligned() : value{} {}
    explicit CacheLineAligned(T v) : value(v) {}

    // 隐式转换
    operator T() const { return value; }
    auto operator=(T v) -> CacheLineAligned& {
        value = v;
        return *this;
    }
};

} // namespace arm64lab
```

### 原子操作 API (参考 ncnn 简洁风格)
```cpp
namespace arm64lab {

// 内存序枚举
enum class MemoryOrder {
    Relaxed,
    Acquire,
    Release,
    AcqRel,
    SeqCst,
};

template<typename T>
class Atomic {
public:
    explicit Atomic(T val = 0) : value_(val) {}

    // 简洁的 API 设计
    auto load(MemoryOrder order = MemoryOrder::SeqCst) const -> T;
    auto store(T val, MemoryOrder order = MemoryOrder::SeqCst) -> void;
    auto exchange(T val, MemoryOrder order = MemoryOrder::SeqCst) -> T;
    auto compare_exchange_strong(T& expected, T desired,
                                  MemoryOrder success = MemoryOrder::SeqCst,
                                  MemoryOrder failure = MemoryOrder::SeqCst) -> bool;
    auto fetch_add(T delta, MemoryOrder order = MemoryOrder::SeqCst) -> T;

    // 运算符重载
    auto operator++() -> T { return fetch_add(1) + 1; }
    auto operator--() -> T { return fetch_sub(1) - 1; }

private:
    mutable T value_;
};

} // namespace arm64lab
```

**C++ 设计:**

```cpp
// include/atomic.hpp
namespace arm64lab {

// 内存序
enum class MemoryOrder {
    Relaxed,
    Acquire,
    Release,
    AcqRel,
    SeqCst,
};

template<typename T>
class Atomic {
public:
    explicit Atomic(T val = 0) : value_(val) {}

    auto load(MemoryOrder order = MemoryOrder::SeqCst) const -> T;
    void store(T val, MemoryOrder order = MemoryOrder::SeqCst);

    auto exchange(T val, MemoryOrder order = MemoryOrder::SeqCst) -> T;
    auto compare_exchange_strong(T& expected, T desired,
                                  MemoryOrder success = MemoryOrder::SeqCst,
                                  MemoryOrder failure = MemoryOrder::SeqCst) -> bool;
    auto compare_exchange_weak(T& expected, T desired,
                                MemoryOrder success = MemoryOrder::SeqCst,
                                MemoryOrder failure = MemoryOrder::SeqCst) -> bool;

    auto fetch_add(T delta, MemoryOrder order = MemoryOrder::SeqCst) -> T;
    auto fetch_sub(T delta, MemoryOrder order = MemoryOrder::SeqCst) -> T;

    auto operator++() -> T { return fetch_add(1) + 1; }
    auto operator--() -> T { return fetch_sub(1) - 1; }

private:
    mutable T value_;
};

// 特化实现
template<>
auto Atomic<u32>::load(MemoryOrder order) const -> u32 {
    u32 result;
    asm volatile("ldar %w0, [%1]"
                 : "=r"(result)
                 : "r"(&value_)
                 : "memory");
    return result;
}

template<>
void Atomic<u32>::store(u32 val, MemoryOrder order) {
    asm volatile("stlr %w0, [%1]"
                 :
                 : "r"(val), "r"(&value_)
                 : "memory");
}

// CAS 实现
template<typename T>
auto Atomic<T>::compare_exchange_strong(T& expected, T desired,
                                         MemoryOrder, MemoryOrder) -> bool {
    u32 success;
    asm volatile("1: ldxr %w0, [%2]\n"
                 "   cmp %w0, %w3\n"
                 "   b.ne 2f\n"
                 "   stxr %w1, %w4, [%2]\n"
                 "   cbnz %w1, 1b\n"
                 "   mov %w1, #1\n"
                 "   b 3f\n"
                 "2: mov %w1, #0\n"
                 "3:"
                 : "=&r"(expected), "=&r"(success)
                 : "r"(&value_), "r"(expected), "r"(desired)
                 : "memory", "cc");
    return success;
}

} // namespace arm64lab
```
